```python

psd_vals = None

alpha_high = None

    global psd_freqs, psd_vals, psd_event, alpha_low, alpha_high, last_psd_vals

    vals = np.array([psd.get_psd(0, i) for i in range(psd.get_frequencies_count())])

    psd_vals = vals
    last_psd_vals = vals

    try:

    except:
```
